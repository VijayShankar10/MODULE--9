## Program:
```cpp
void rightrotate(node* temp)
{
    node* left = temp->l;
    temp->l = left->r;
    
    if (temp->l)
        temp->l->p = temp;
        
    left->p = temp->p;
    
    if (!temp->p)
        root = left;
    else if (temp == temp->p->l)
        temp->p->l = left;
    else
        temp->p->r = left;
        
    left->r = temp;
    temp->p = left;
}
```
